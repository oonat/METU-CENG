\begin{equation*}
\begin{split}

\end{equation*}

\begin{equation*}

\end{split}
\end{equation*}

\begin{equation*}
\begin{split}
x^{p-1} = x^{qy + r} = (x^{y})^{q}x^{r}
\end{split}
\end{equation*}
